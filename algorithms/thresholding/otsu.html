Otsuз®—жі•

Otsuз®—жі•жҳҜдёҖз§ҚиҮӘеҠЁйҳҲеҖјйҖүжӢ©ж–№жі•пјҢйҖҡиҝҮжңҖеӨ§еҢ–зұ»й—ҙж–№е·®жқҘзЎ®е®ҡжңҖдјҳйҳҲеҖјпјҢйҖӮз”ЁдәҺеҸҢеі°зӣҙж–№еӣҫзҡ„еӣҫеғҸдәҢеҖјеҢ–гҖӮ

з®—жі•еҺҹзҗҶ

Otsuз®—жі•з”ұж—Ҙжң¬еӯҰиҖ…Nobuyuki OtsuеңЁ1979е№ҙжҸҗеҮәпјҢжҳҜдёҖз§ҚиҮӘеҠЁзЎ®е®ҡйҳҲеҖјзҡ„ж–№жі•гҖӮиҜҘз®—жі•еҹәдәҺеӣҫеғҸзҡ„зҒ°еәҰзӣҙж–№еӣҫпјҢйҖҡиҝҮжңҖеӨ§еҢ–еүҚжҷҜе’ҢиғҢжҷҜд№Ӣй—ҙзҡ„зұ»й—ҙж–№е·®жқҘзЎ®е®ҡжңҖдјҳйҳҲеҖјгҖӮ

з®—жі•зҡ„ж ёеҝғжҖқжғіжҳҜжүҫеҲ°дёҖдёӘйҳҲеҖјTпјҢдҪҝеҫ—еүҚжҷҜе’ҢиғҢжҷҜд№Ӣй—ҙзҡ„зұ»й—ҙж–№е·®жңҖеӨ§пјҢжҲ–иҖ…зӯүд»·ең°пјҢдҪҝеҫ—еүҚжҷҜе’ҢиғҢжҷҜзҡ„зұ»еҶ…ж–№е·®жңҖе°ҸгҖӮ

и®ҫеӣҫеғҸзҡ„зҒ°еәҰзә§дёә[0, L-1]пјҢз¬¬iдёӘзҒ°еәҰзә§зҡ„жҰӮзҺҮдёәp_iпјҢеҲҷпјҡ

ПүвӮҖ(T) = ОЈбөўвӮҢвӮҖбөҖвҒ»В№ pбөў (иғҢжҷҜеғҸзҙ жҜ”дҫӢ)

ПүвӮҒ(T) = ОЈбөўвӮҢбөҖбҙёвҒ»В№ pбөў (еүҚжҷҜеғҸзҙ жҜ”дҫӢ)

ОјвӮҖ(T) = (ОЈбөўвӮҢвӮҖбөҖвҒ»В№ i Г— pбөў) / ПүвӮҖ(T) (иғҢжҷҜе№іеқҮзҒ°еәҰ)

ОјвӮҒ(T) = (ОЈбөўвӮҢбөҖбҙёвҒ»В№ i Г— pбөў) / ПүвӮҒ(T) (еүҚжҷҜе№іеқҮзҒ°еәҰ)

зұ»й—ҙж–№е·® ПғВІбөҰ(T) = ПүвӮҖ(T) Г— ПүвӮҒ(T) Г— [ОјвӮҖ(T) - ОјвӮҒ(T)]ВІ

import cv2
import numpy as np
import matplotlib.pyplot as plt

def otsu_threshold_builtin(image_path):
    """
    дҪҝз”ЁOpenCVеҶ…зҪ®еҮҪж•°е®һзҺ°OtsuйҳҲеҖј
    :param image_path: иҫ“е…ҘеӣҫеғҸи·Ҝеҫ„
    :return: еҺҹеӣҫе’ҢOtsuдәҢеҖјеҢ–еҗҺзҡ„еӣҫеғҸ
    """
    # иҜ»еҸ–еӣҫеғҸ
    img = cv2.imread(image_path)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    
    # еә”з”ЁOtsuйҳҲеҖј
    _, binary = cv2.threshold(gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    
    # иҺ·еҸ–и®Ўз®—еҮәзҡ„йҳҲеҖј
    threshold_value = cv2.threshold(gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)[0]
    
    return img, cv2.cvtColor(binary, cv2.COLOR_GRAY2BGR), threshold_value

def manual_otsu_threshold(image_path):
    """
    жүӢеҠЁе®һзҺ°OtsuйҳҲеҖјз®—жі•
    :param image_path: иҫ“е…ҘеӣҫеғҸи·Ҝеҫ„
    :return: дәҢеҖјеҢ–еҗҺзҡ„еӣҫеғҸе’Ңи®Ўз®—еҮәзҡ„йҳҲеҖј
    """
    # иҜ»еҸ–еӣҫеғҸ
    img = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE)
    
    # и®Ўз®—зӣҙж–№еӣҫ
    hist, _ = np.histogram(img.flatten(), 256, [0, 256])
    
    # еҪ’дёҖеҢ–зӣҙж–№еӣҫ
    hist_norm = hist.astype(float) / hist.sum()
    
    # еҲқе§ӢеҢ–еҸҳйҮҸ
    max_variance = 0
    optimal_threshold = 0
    
    # йҒ°еәҰзә§жҖ»ж•°
    total_levels = len(hist_norm)
    
    # и®Ўз®—еӣҫеғҸзҡ„жҖ»дҪ“е№іеқҮзҒ°еәҰ
    total_mean = sum(i * hist_norm[i] for i in range(total_levels))
    
    # еҲқе§ӢеҢ–иғҢжҷҜжҰӮзҺҮе’Ңе№іеқҮзҒ°еәҰ
    omega_0 = 0  # иғҢжҷҜеғҸзҙ жҜ”дҫӢ
    mu_0 = 0     # иғҢжҷҜе№іеқҮзҒ°еәҰ
    
    # йҒ°еәҰзә§йҒҚеҺҶ
    for t in range(total_levels):
        # жӣҙж–°иғҢжҷҜжҰӮзҺҮ
        omega_0 += hist_norm[t]
        
        # жӣҙж–°иғҢжҷҜе№іеқҮзҒ°еәҰ
        if omega_0 > 0:
            mu_0 += t * hist_norm[t]
            mu_0_t = mu_0 / omega_0 if omega_0 > 0 else 0
        else:
            mu_0_t = 0
        
        # и®Ўз®—еүҚжҷҜжҰӮзҺҮе’Ңе№іеқҮзҒ°еәҰ
        omega_1 = 1.0 - omega_0
        if omega_1 > 0:
            mu_1_t = (total_mean - omega_0 * mu_0_t) / omega_1
        else:
            mu_1_t = 0
        
        # и®Ўз®—зұ»й—ҙж–№е·®
        between_class_variance = omega_0 * omega_1 * (mu_0_t - mu_1_t) ** 2
        
        # жӣҙж–°жңҖдјҳйҳҲеҖј
        if between_class_variance > max_variance:
            max_variance = between_class_variance
            optimal_threshold = t
    
    # еә”з”Ёи®Ўз®—еҮәзҡ„йҳҲеҖјиҝӣиЎҢдәҢеҖјеҢ–
    binary = np.zeros_like(img)
    binary[img > optimal_threshold] = 255
    binary[img <= optimal_threshold] = 0
    
    return cv2.cvtColor(img, cv2.COLOR_GRAY2BGR), cv2.cvtColor(binary, cv2.COLOR_GRAY2BGR), optimal_threshold

def otsu_with_histogram(image_path):

def multi_level_otsu(image_path, levels=3):
    """
    еӨҡзә§OtsuйҳҲеҖјпјҲе°ҶеӣҫеғҸеҲҶдёәеӨҡдёӘзӯүзә§пјү
    :param image_path: иҫ“е…ҘеӣҫеғҸи·Ҝеҫ„
    :param levels: еҲҶеүІзӯүзә§ж•°
    :return: еҺҹеӣҫе’ҢеӨҡзә§еҲҶеүІз»“жһң
    """
    # иҜ»еҸ–еӣҫеғҸ
    img = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE)
    
    # дҪҝз”ЁOpenCVзҡ„еӨҡзә§OtsuйҳҲеҖј
    if levels == 2:
        _, result = cv2.threshold(img, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    elif levels == 3:
        _, result = cv2.threshold(img, 0, 255, cv2.THRESH_TRIANGLE)
    else:
        # еҜ№дәҺжӣҙеӨҡзӯүзә§пјҢжҲ‘д»¬еҸҜд»ҘдҪҝз”ЁеҲҶеұӮж–№жі•
        # иҝҷйҮҢз®ҖеҢ–еӨ„зҗҶпјҢдҪҝз”ЁеқҮеҢҖеҲҶеүІ
        step = 255 // levels
        result = (img // step) * step
    
    return cv2.cvtColor(img, cv2.COLOR_GRAY2BGR), cv2.cvtColor(result, cv2.COLOR_GRAY2BGR)

def compare_thresholding_methods(image_path):
    """
    жҜ”иҫғдёҚеҗҢйҳҲеҖјж–№жі•зҡ„ж•Ҳжһң
    :param image_path: иҫ“е…ҘеӣҫеғҸи·Ҝеҫ„
    :return: еҺҹеӣҫе’ҢдёҚеҗҢж–№жі•зҡ„дәҢеҖјеҢ–з»“жһң
    """
    # иҜ»еҸ–еӣҫеғҸ
    img = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE)
    
    # дёҚеҗҢзҡ„йҳҲеҖјж–№жі•
    methods = {
        'Global (127)': cv2.threshold(img, 127, 255, cv2.THRESH_BINARY)[1],
        'Otsu': cv2.threshold(img, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)[1],
        'Adaptive Mean': cv2.adaptiveThreshold(img, 255, cv2.ADAPTIVE_THRESH_MEAN_C, cv2.THRESH_BINARY, 11, 2),
        'Adaptive Gaussian': cv2.adaptiveThreshold(img, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY, 11, 2),
        'Triangle': cv2.threshold(img, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_TRIANGLE)[1]
    }
    
    results = {}
    for name, result in methods.items():
        results[name] = cv2.cvtColor(result, cv2.COLOR_GRAY2BGR)
    
    return cv2.cvtColor(img, cv2.COLOR_GRAY2BGR), results

def otsu_performance_analysis(image_path):
    """
    Otsuз®—жі•жҖ§иғҪеҲҶжһҗ
    :param image_path: иҫ“е…ҘеӣҫеғҸи·Ҝеҫ„
    :return: еҲҶжһҗз»“жһң
    """
    # иҜ»еҸ–еӣҫеғҸ
    img = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE)
    
    # и®Ўз®—зӣҙж–№еӣҫ
    hist, _ = np.histogram(img.flatten(), 256, [0, 256])
    hist_norm = hist.astype(float) / hist.sum()
    
    # и®Ўз®—OtsuйҳҲеҖј
    total_mean = sum(i * hist_norm[i] for i in range(256))
    omega_0 = 0
    mu_0 = 0
    max_variance = 0
    optimal_threshold = 0
    
    for t in range(256):
        omega_0 += hist_norm[t]
        if omega_0 > 0:
            mu_0 += t * hist_norm[t]
            mu_0_t = mu_0 / omega_0 if omega_0 > 0 else 0
        else:
            mu_0_t = 0
        
        omega_1 = 1.0 - omega_0
        if omega_1 > 0:
            mu_1_t = (total_mean - omega_0 * mu_0_t) / omega_1
        else:
            mu_1_t = 0
        
        between_class_variance = omega_0 * omega_1 * (mu_0_t - mu_1_t) ** 2
        
        if between_class_variance > max_variance:
            max_variance = between_class_variance
            optimal_threshold = t
    
    # и®Ўз®—еҲҶеүІиҙЁйҮҸжҢҮж Ү
    foreground_pixels = np.sum(img > optimal_threshold)
    background_pixels = np.sum(img <= optimal_threshold)
    total_pixels = img.size
    
    foreground_ratio = foreground_pixels / total_pixels
    background_ratio = background_pixels / total_pixels
    
    # еә”з”ЁйҳҲеҖј
    binary = np.zeros_like(img)
    binary[img > optimal_threshold] = 255
    binary[img <= optimal_threshold] = 0
    
    return {
        'optimal_threshold': optimal_threshold,
        'max_between_class_variance': max_variance,
        'foreground_ratio': foreground_ratio,
        'background_ratio': background_ratio,
        'original_image': cv2.cvtColor(img, cv2.COLOR_GRAY2BGR),
        'binary_image': cv2.cvtColor(binary, cv2.COLOR_GRAY2BGR)
    }
